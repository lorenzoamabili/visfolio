main()

The if name == ‘main’: main() construct is a common pattern used in Python to ensure that a block of code is only executed when the script is run directly as the main module, and not when it is imported as a module by another script.

        self.name = name

    def greet(self):
        print(f"Hello, {self.name}!")

    def main(self):

        self.greet()

if __name__ == '__main__':
    my_object = MyClass("Alice")

class MyClass:
    n = 3
    m = 5

    def __init__(self, name):
        self.name = name

    def greet(self):
        print(f"Hello, {self.name}!")

    def calculate_sum(self, a, b):
        return a + b

    def calculate_product(self, a, b):
        return a * b

    def print_sum_and_product(self, a, b):
        sum_result = self.calculate_sum(a, b)
        product_result = self.calculate_product(a, b)
        print(f"The sum of {a} and {b} is: {sum_result}")
        print(f"The product of {a} and {b} is: {product_result}")

    def main(self):
        # Main logic of the script
        self.greet()
        self.print_sum_and_product(self.n, self.m)


if __name__ == '__main__':
    my_object = MyClass("Alice")
    my_object.main()
